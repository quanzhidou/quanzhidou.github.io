豆豆的刷题日记（9）——双指针之滑动窗口

leetcode 438题找到所有字母异位词

思路

和leetcode 567题字符串的排列的思路一样，只需设置一个结果数组，在567题返回结果为true时不直接返回，而是记录到结果数组里即可

go语言代码

func findAnagrams(s string, p string) []int {
	left, right := 0, 0
	window, need := make(map[byte]int), make(map[byte]int)
	valid := 0
	var start []int
	for i := range p {
		need[p[i]]++
	}
	for right < len(s) {
		c := s[right]
		right++
		if need[c] != 0 {
			window[c]++
			if window[c] == need[c] {
				valid++
			}
		}
		for right-left >= len(p) {
			if valid == len(need) {
                //这里不直接返回，记录到结果数组里
				start = append(start, left)
			}
			d := s[left]
			left++
			if need[d] != 0 {
				if window[d] == need[d] {
					valid--
				}
				window[d]--
			}
		}
	}
    //最后返回结果数组
	return start
}
